function helloWorldHtmlServic(){
  
  var ui = DocumentApp.getUi();
  var html = HtmlService.createTemplateFromFile('sideBar').evaluate()
      .setTitle('Sidebar Example').setWidth(300)
      .setSandboxMode(HtmlService.SandboxMode.NATIVE);

  ui.showSidebar(html);
}
